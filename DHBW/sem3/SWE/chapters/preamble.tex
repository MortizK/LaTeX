    #3

    \textbf{Vorbedingungen:}\\
    #4

    \ifx&#5&  % Prüft, ob #5 leer ist
    \else
    \textbf{Ablauf:}
        \begin{enumerate}
            #5
        \end{enumerate}
    \fi

    \textbf{Nachbedingungen:}\\
    #6

    \end{addmargin}
}
